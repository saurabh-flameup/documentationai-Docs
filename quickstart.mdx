## Quick Start Guide

This guide will walk you through integrating Flare into your application. By the end, you'll be able to identify users, track events, and send your first message.

<Callout kind="info">**Prerequisites**: You'll need a Flare account and an API key. Sign up at [app.flameup.ai](https://app.flameup.ai) if you haven't already.</Callout>

## Step 1: Get Your API Key

<Steps>
  <Step title="Log into Dashboard" icon="user">
    Go to [app.flameup.ai](https://app.flameup.ai) and log in to your account.
  </Step>
  <Step title="Navigate to API Keys" icon="key">
    Go to **Settings** > **API Keys** in the sidebar.
  </Step>
  <Step title="Create a New Key" icon="plus">
    Click **Create API Key**, give it a name like "My App - Development", select **Test** environment, and grant the following permissions:
    - `people:read`
    - `people:write`
    - `events:read`
    - `events:write`
  </Step>
  <Step title="Copy Your Key" icon="copy">
    Copy the API key and save it securely. It starts with `flare_test_...`
  </Step>
</Steps>

## Step 2: Identify a User

When a user signs up or logs in, identify them to Flare. This creates or updates their profile.

<CodeGroup>
```javascript
const API_KEY = 'flare_test_your_key_here';
const WORKSPACE_ID = 'your_workspace_id';

async function identifyUser(userId, email, traits) {
  const response = await fetch(
    `https://api.flameup.ai/api/v1/workspaces/${WORKSPACE_ID}/people`,
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'X-API-Key': API_KEY
      },
      body: JSON.stringify({
        userId: userId,
        email: email,
        traits: traits
      })
    }
  );

  return response.json();
}

// Example usage
const user = await identifyUser(
  'user_12345',
  'sarah@example.com',
  {
    first_name: 'Sarah',
    last_name: 'Connor',
    plan: 'premium',
    signup_date: '2024-01-15'
  }
);

console.log('User identified:', user.id);
```

```python
import requests

API_KEY = 'flare_test_your_key_here'
WORKSPACE_ID = 'your_workspace_id'

def identify_user(user_id, email, traits):
    response = requests.post(
        f'https://api.flameup.ai/api/v1/workspaces/{WORKSPACE_ID}/people',
        headers={
            'Content-Type': 'application/json',
            'X-API-Key': API_KEY
        },
        json={
            'userId': user_id,
            'email': email,
            'traits': traits
        }
    )
    return response.json()

# Example usage
user = identify_user(
    'user_12345',
    'sarah@example.com',
    {
        'first_name': 'Sarah',
        'last_name': 'Connor',
        'plan': 'premium',
        'signup_date': '2024-01-15'
    }
)

print(f"User identified: {user['id']}")
```

```bash
curl -X POST "https://api.flameup.ai/api/v1/workspaces/{workspace_id}/people" \
  -H "Content-Type: application/json" \
  -H "X-API-Key: flare_test_your_key_here" \
  -d '{
    "userId": "user_12345",
    "email": "sarah@example.com",
    "traits": {
      "first_name": "Sarah",
      "last_name": "Connor",
      "plan": "premium",
      "signup_date": "2024-01-15"
    }
  }'
```
</CodeGroup>

<Callout kind="tip">The `userId` is your unique identifier for this user. It can be a database ID, username, or any stable identifier in your system.</Callout>

## Step 3: Track an Event

Track user actions to understand behavior and trigger campaigns.

<CodeGroup>
```javascript
async function trackEvent(userId, eventName, properties) {
  const response = await fetch(
    `https://api.flameup.ai/api/v1/workspaces/${WORKSPACE_ID}/events`,
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'X-API-Key': API_KEY
      },
      body: JSON.stringify({
        userId: userId,
        event: eventName,
        properties: properties,
        timestamp: new Date().toISOString()
      })
    }
  );

  return response.json();
}

// Track a purchase
await trackEvent('user_12345', 'purchase_completed', {
  order_id: 'order_789',
  amount: 99.99,
  currency: 'USD',
  items: ['product_a', 'product_b']
});

console.log('Event tracked!');
```

```python
from datetime import datetime

def track_event(user_id, event_name, properties):
    response = requests.post(
        f'https://api.flameup.ai/api/v1/workspaces/{WORKSPACE_ID}/events',
        headers={
            'Content-Type': 'application/json',
            'X-API-Key': API_KEY
        },
        json={
            'userId': user_id,
            'event': event_name,
            'properties': properties,
            'timestamp': datetime.utcnow().isoformat() + 'Z'
        }
    )
    return response.json()

# Track a purchase
track_event('user_12345', 'purchase_completed', {
    'order_id': 'order_789',
    'amount': 99.99,
    'currency': 'USD',
    'items': ['product_a', 'product_b']
})

print('Event tracked!')
```

```bash
curl -X POST "https://api.flameup.ai/api/v1/workspaces/{workspace_id}/events" \
  -H "Content-Type: application/json" \
  -H "X-API-Key: flare_test_your_key_here" \
  -d '{
    "userId": "user_12345",
    "event": "purchase_completed",
    "properties": {
      "order_id": "order_789",
      "amount": 99.99,
      "currency": "USD",
      "items": ["product_a", "product_b"]
    },
    "timestamp": "2024-01-15T10:30:00Z"
  }'
```
</CodeGroup>

## Step 4: Register a Device (Optional)

If your app uses push notifications, register the device token:

<CodeGroup>
```javascript
async function registerDevice(userId, deviceToken, platform) {
  const response = await fetch(
    `https://api.flameup.ai/api/v1/workspaces/${WORKSPACE_ID}/devices`,
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'X-API-Key': API_KEY
      },
      body: JSON.stringify({
        user_id: userId,
        device_token: deviceToken,
        platform: platform, // 'ios', 'android', or 'web'
        device_name: 'iPhone 15 Pro',
        app_version: '2.1.0'
      })
    }
  );

  return response.json();
}

// Register an iOS device
await registerDevice(
  'user_12345',
  'fcm_device_token_here',
  'ios'
);
```

```python
def register_device(user_id, device_token, platform):
    response = requests.post(
        f'https://api.flameup.ai/api/v1/workspaces/{WORKSPACE_ID}/devices',
        headers={
            'Content-Type': 'application/json',
            'X-API-Key': API_KEY
        },
        json={
            'user_id': user_id,
            'device_token': device_token,
            'platform': platform,  # 'ios', 'android', or 'web'
            'device_name': 'iPhone 15 Pro',
            'app_version': '2.1.0'
        }
    )
    return response.json()

# Register an iOS device
register_device(
    'user_12345',
    'fcm_device_token_here',
    'ios'
)
```
</CodeGroup>

## Step 5: Send a Transactional Message

Send a one-off message to a user (e.g., order confirmation):

<CodeGroup>
```javascript
async function sendTransactional(userId, templateId, data) {
  const response = await fetch(
    `https://api.flameup.ai/api/v1/workspaces/${WORKSPACE_ID}/transactional/send`,
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'X-API-Key': API_KEY
      },
      body: JSON.stringify({
        user_id: userId,
        template_id: templateId,
        data: data
      })
    }
  );

  return response.json();
}

// Send order confirmation
await sendTransactional('user_12345', 'order_confirmation', {
  order_id: 'order_789',
  total: '$99.99',
  items: '2 items'
});
```

```python
def send_transactional(user_id, template_id, data):
    response = requests.post(
        f'https://api.flameup.ai/api/v1/workspaces/{WORKSPACE_ID}/transactional/send',
        headers={
            'Content-Type': 'application/json',
            'X-API-Key': API_KEY
        },
        json={
            'user_id': user_id,
            'template_id': template_id,
            'data': data
        }
    )
    return response.json()

# Send order confirmation
send_transactional('user_12345', 'order_confirmation', {
    'order_id': 'order_789',
    'total': '$99.99',
    'items': '2 items'
})
```
</CodeGroup>

## Verification

Check that everything is working:

1. **Dashboard**: Log into [app.flameup.ai](https://app.flameup.ai) and check the People section - you should see your test user.

2. **API**: Fetch the user you just created:

```bash
curl "https://api.flameup.ai/api/v1/workspaces/{workspace_id}/people/search?email=sarah@example.com" \
  -H "X-API-Key: flare_test_your_key_here"
```
